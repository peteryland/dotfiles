## Fix screen tearing issue on NVidia cards:

# /etc/X11/xorg.conf.d/20-nvidia.conf

Section "Device"
  Identifier "NVidia Card"
  Driver     "nvidia"
EndSection

Section "Screen"
  Identifier "Screen0"
  Option     "MetaModes" "nvidia-auto-select +0+0 {ForceFullCompositionPipeline=On}"
